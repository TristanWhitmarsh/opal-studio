# Setup:
#   conda create -n opal-mesmer python=3.9
#   conda activate opal-mesmer
#   pip install -r requirements2.txt

# Known risks:
#   - numpy 1.24.x may hit deprecated APIs in TF 2.8 at runtime (watch for warnings)
#   - deepcell only supports Python <3.11, use Python 3.9

#   - instanseg-torch may require torch >= 2.x; watch for conflicts
